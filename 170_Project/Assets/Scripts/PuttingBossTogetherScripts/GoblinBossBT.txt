tree("Root")
	while IsAlive
		while CheckDialogueStatus
			while IsCooldownOver
				random
					tree("Melee")
					tree("SteadyAimFire")
					tree("GetEmBoys")
					tree("SurroundEm")

tree("SurroundEm")
	sequence
		CheckAttacks("surroundEm")
		SameAsLastAttack("surroundEm")
		mute
			while CheckStartup
				IncrementStartup
		SurroundEm
		ResetStartup
		ResetCooldown
