tree("Root")
	while IsAlive
		while CheckDialogueStatus
			fallback
				while CheckWaitingStatus
					sequence
						Wait 3.0
						ToggleWaitingStatus
				while IsCooldownOver
					random
						//tree("Melee")
						tree("SteadyAimFire")
						tree("GetEmBoys")
						tree("SurroundEm")
						tree("PickaxeThrow")
						tree("GoblinCharge")

tree("SurroundEm")
	sequence
		CheckAttacks("surroundEm")
		SameAsLastAttack("surroundEm")
		AnnounceAttack("surroundEm")
		ResetStartup
		mute
			while CheckStartup
				IncrementStartup
		SurroundEm
		ResetCooldown
